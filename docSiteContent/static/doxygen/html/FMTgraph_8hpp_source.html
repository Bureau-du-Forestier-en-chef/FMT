319 std::queue<FMTvertex_descriptor> initialize(const std::vector<Core::FMTactualdevelopment>& actdevelopments)
320 {
321 std::queue<FMTvertex_descriptor>actives;
322 try {
323 developments.clear();
324 const int startingperiod = actdevelopments.begin()->getperiod();
325 const int actualperiod = startingperiod+1;
326 FMTvertex_iterator vertex_iterator, vertex_iterator_end;
327 //End always stay the same use .end() for non valid period
328 boost::tie(vertex_iterator, vertex_iterator_end) = boost::vertices(data);
329 for (int period = 0 ; period <= actualperiod;++period)
330 {
331 developments.push_back(FMTvertex_pair(vertex_iterator_end,vertex_iterator_end));
332 }
333 const int constraint_id = -1;
334 int edge_id = -1;
335 const double proportion = 100;
336 stats.edges = 0;
337 stats.vertices = 0;
338 std::vector<FMTvertex_descriptor>P0descriptors;
339 P0descriptors.reserve(actdevelopments.size());
340 for (const Core::FMTactualdevelopment& development : actdevelopments)
341 {
342 const FMTvertexproperties properties(development, constraint_id);
343 const FMTvertex_descriptor newvertex = boost::add_vertex(properties, data);
344 P0descriptors.push_back(newvertex);
345 //++vertex_iterator;
346 ++stats.vertices;
347 }
348 size_t poid = 0;
349 boost::unordered_set<Core::FMTlookup<FMTvertex_descriptor, Core::FMTdevelopment>> devsets;
350 for (const Core::FMTactualdevelopment& development : actdevelopments)
351 {
352 //P1
353 Core::FMTfuturdevelopment P1dev(development);
354 P1dev.setperiod(actualperiod);
355 const FMTvertex_descriptor tovertex = adddevelopment(P1dev, devsets);
356 actives.push(tovertex);
357 //Now set the edge!!
