109 void presolvelist(const FMTmaskfilter& filter,
110 const std::vector<FMTtheme>& originalthemes,
111 const std::vector<FMTtheme>& newthemes)
112 {
113 try {
114 if (!canshrink())
115 {
116 unshrink(originalthemes);
117 }
118 const std::vector<FMTtheme>maskthemes = filter.getselectedthemes(originalthemes);
119 std::vector<std::pair<FMTmask, T>>newdata;

122 if (filter.canpresolve(object.first,maskthemes))

260 // DocString: FMTlist::findsetswithfiltered
265 std::vector<const T*> findsetswithfiltered(const FMTmask& newkey) const
266 {
267 std::vector<const T*>allhits;
268 boost::unordered_map<FMTmask, std::vector<int>>::const_iterator fast_it = fastpass.find(newkey);
269 if (fast_it != fastpass.end())
270 {
271 allhits.reserve(fast_it->second.size());
272 for (const int& location : fast_it->second)
273 {
274 allhits.push_back(&data.at(location).second);
275 }
276 }
277 else {
278 fastpass[newkey] = std::vector<int>();
279 int location = 0;
280 for (const std::pair<FMTmask, T>& object : data)
281 {
282 if (newkey.issubsetof(object.first))
283 {
284 fastpass[newkey].push_back(location);
285 allhits.push_back(&object.second);
286 }
287 ++location;
288 }
289 fastpass[newkey].shrink_to_fit();
290 }
291 return allhits;
292 }

301 // DocString: FMTlist::shrink
305 void shrink()
306 {
307 try {
308 fastpass.clear();
309 std::vector<Core::FMTmask> filteredmasks;
310 for (const std::pair<FMTmask, T>& object : data)
311 {
312 filteredmasks.push_back(object.first);
313 }
314 if (filteredmasks.empty())
315 {
316 _exhandler->raise(Exception::FMTexc::FMTinvalid_maskrange,"Empty mask", "FMTactionparser::shrink", __LINE__, __FILE__);
317 }
318 filter = Core::FMTmaskfilter(filteredmasks);
319 for (std::pair<FMTmask, T>& object : data)
320 {
321 object.first = filter.filter(object.first);
322 }
323 data.shrink_to_fit();
